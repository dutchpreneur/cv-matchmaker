    <script>

        async function uploadVacancy() {

            const vacancyFile = document.getElementById('vacancyFile').files[0];

            if (!vacancyFile) {

                alert('Please select a vacancy file');

                return;

            }

           

            const formData = new FormData();

            formData.append('file', vacancyFile);

 

            try {

                const response = await fetch('/upload_vacancy', {

                    method: 'POST',

                    body: formData

                });

                const result = await response.json();

                alert(result.message);

            } catch (error) {

                console.error('Error:', error);

                alert('An error occurred while uploading the vacancy');

            }

        }

 

        async function uploadCVs() {

            const cvFiles = document.getElementById('cvFiles').files;

            if (cvFiles.length === 0) {

                alert('Please select at least one CV file');

                return;

            }

           

            const formData = new FormData();

            for (let i = 0; i < cvFiles.length; i++) {

                formData.append('files', cvFiles[i]);

            }

 

            try {

                const response = await fetch('/upload_cvs', {

                    method: 'POST',

                    body: formData

                });

                const result = await response.json();

                alert(result.message);

            } catch (error) {

                console.error('Error:', error);

                alert('An error occurred while uploading the CVs');

            }

        }
